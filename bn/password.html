আপনার পাসওয়ার্ড কি ফাঁস হয়েছে?


ডেটা লঙ্ঘনে উন্মোচিত 835,955,029টি পাসওয়ার্ডের বিরুদ্ধে আপনার পাসওয়ার্ড পরীক্ষা করুন। বিনামূল্যে, গোপনীয় এবং তাৎক্ষণিক।


আপনার পাসওয়ার্ড কখনও এই পেজ ছেড়ে যায় না। আপনার ব্রাউজার এটি হ্যাশ করে এবং হ্যাশের প্রথম 10টি অক্ষর পাঠায়।

Xposed Passwords ফাঁস পরীক্ষক কী?

আপনার পাসওয়ার্ড কোনো ডেটা লঙ্ঘনে ফাঁস হয়েছে কিনা তা জানতে Xposed Passwords হলো আপনার বিনামূল্যের পাসওয়ার্ড নিরাপত্তা পরীক্ষক। আমরা যা করি তা এখানে দেওয়া হলো:

✓ আমরা ডেটা লঙ্ঘনে ফাঁস হওয়া ৮৩৫+ মিলিয়ন পাসওয়ার্ডের একটি নিরাপদ ডেটাবেস রক্ষণাবেক্ষণ করি

✓ আপনি যখন কোনো পাসওয়ার্ড যাচাই করেন, আপনার ব্রাউজার সেটিকে SHA-3 (Keccak-512) দিয়ে লোকালভাবে হ্যাশ করে এবং সেই হ্যাশের কেবল প্রথম 10টি অক্ষর পাঠায়। আপনার পাসওয়ার্ড কখনও আপনার ডিভাইস ছেড়ে যায় না।

✓ আমরা সেই আংশিক হ্যাশ k-অ্যানোনিমিটির মাধ্যমে আমাদের ডেটাবেসের সাথে মেলাই, তাই আপনার পাসওয়ার্ড না দেখেই মিল সম্পর্কে জানাতে পারি। আপনি যা লেখেন তা সংরক্ষণ বা লগ করা হয় না।

আমাদের আপনার পাসওয়ার্ড ফাঁস পরীক্ষক হিসেবে ভাবুন, উন্মোচিত পাসওয়ার্ডের জন্য একটি ধাতু শনাক্তকারীর মতো। আপনার পাসওয়ার্ড যদি আমাদের ডেটাবেসে পাওয়া যায়, তবে এর অর্থ হ্যাকাররা ইতিমধ্যেই এটি জানতে পারে, এবং আপনার এটি অবিলম্বে পরিবর্তন করা উচিত।

আরও জানতে চান? আমাদের বিস্তারিত নির্দেশিকা দেখুন Xposed Passwords কীভাবে আপনাকে নিরাপদ রাখে সে সম্পর্কে।

from Crypto.Hash import keccak
import requests
import urllib.parse

def check_password(password: str) -> str:
    if not password:
        return "Please enter a password to check"
    
    try:
        # Create SHA3-512 (Keccak) hash
        k = keccak.new(digest_bits=512)
        k.update(password.encode())
        pwd_hash = k.hexdigest()[:10]
        
        # Call API
        api_url = f'https://passwords.xposedornot.com/api/v1/pass/anon/{urllib.parse.quote(pwd_hash)}'
        response = requests.get(api_url)
        
        if response.status_code == 200:
            return "Warning: Password has been exposed in data breaches"
        elif response.status_code == 404:
            return "Good news! Password is safe"
        else:
            return f"Error: Unexpected response (Status code: {response.status_code})"
            
    except Exception as e:
        return f"Error: {str(e)}"

import org.bouncycastle.jcajce.provider.digest.Keccak;
import java.net.HttpURLConnection;
import java.net.URL;
import java.net.URLEncoder;
import java.nio.charset.StandardCharsets;

public class PasswordChecker {
    public static String checkPassword(String password) {
        if (password == null || password.isEmpty()) {
            return "Please enter a password to check";
        }

        try {
            // Create Keccak-512 hash
            Keccak.Digest512 digest = new Keccak.Digest512();
            byte[] hash = digest.digest(password.getBytes(StandardCharsets.UTF_8));
            String pwd_hash = bytesToHex(hash).substring(0, 10);

            // Call API
            String apiUrl = "https://passwords.xposedornot.com/api/v1/pass/anon/" + 
                           URLEncoder.encode(pwd_hash, StandardCharsets.UTF_8.toString());
            URL url = new URL(apiUrl);
            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
            conn.setRequestMethod("GET");

            int responseCode = conn.getResponseCode();
            if (responseCode == 200) {
                return "Warning: Password has been exposed in data breaches";
            } else if (responseCode == 404) {
                return "Good news! Password is safe";
            } else {
                return "Error: Unexpected response (Status code: " + responseCode + ")";
            }
        } catch (Exception e) {
            return "Error: " + e.getMessage();
        }
    }

    private static String bytesToHex(byte[] bytes) {
        StringBuilder result = new StringBuilder();
        for (byte b : bytes) {
            result.append(String.format("%02x", b));
        }
        return result.toString();
    }

    public static void main(String[] args) {
        // Example usage
        String result = checkPassword("test-password");
        System.out.println(result);
    }
}
			      
package main

import (
    "encoding/hex"
    "fmt"
    "net/http"
    "net/url"
    "golang.org/x/crypto/sha3"
)

func checkPassword(password string) string {
    if password == "" {
        return "Please enter a password to check"
    }

    // Create SHA3-512 (Keccak) hash
    hash := sha3.NewLegacyKeccak512()
    hash.Write([]byte(password))
    pwd_hash := hex.EncodeToString(hash.Sum(nil))[:10]

    // Call API
    apiUrl := fmt.Sprintf("https://passwords.xposedornot.com/api/v1/pass/anon/%s",
        url.QueryEscape(pwd_hash))
    
    resp, err := http.Get(apiUrl)
    if err != nil {
        return fmt.Sprintf("Error: %v", err)
    }
    defer resp.Body.Close()

    switch resp.StatusCode {
    case http.StatusOK:
        return "Warning: Password has been exposed in data breaches"
    case http.StatusNotFound:
        return "Good news! Password is safe"
    default:
        return fmt.Sprintf("Error: Unexpected response (Status code: %d)", resp.StatusCode)
    }
}

func main() {
    // Example usage
    var password string
    fmt.Print("Enter password to check: ")
    fmt.Scanln(&password)
    
    result := checkPassword(password)
    fmt.Println(result)
}

use tiny_keccak::{Hasher, Keccak};

fn check_password(password: &str) -> String {
    if password.is_empty() {
        return "Please enter a password to check".to_string();
    }

    // Create Keccak-512 hash
    let mut hasher = Keccak::v512();
    let mut output = [0u8; 64];
    hasher.update(password.as_bytes());
    hasher.finalize(&mut output);
    let pwd_hash = hex::encode(&output[..5]); // First 10 hex chars

    // Call API
    let api_url = format!(
        "https://passwords.xposedornot.com/api/v1/pass/anon/{}",
        urlencoding::encode(&pwd_hash)
    );

    match reqwest::blocking::get(&api_url) {
        Ok(response) => match response.status().as_u16() {
            200 => "Warning: Password has been exposed in data breaches".to_string(),
            404 => "Good news! Password is safe".to_string(),
            code => format!("Error: Unexpected response (Status code: {})", code),
        },
        Err(e) => format!("Error: {}", e),
    }
}

আমার পাসওয়ার্ড উন্মোচিত হলে আমার কী করা উচিত?

১. অবিলম্বে আপনার পাসওয়ার্ড পরিবর্তন করুন: যদি আপনি জানতে পারেন যে আপনার পাসওয়ার্ড কোনো ডেটা লঙ্ঘনে আপস করা হয়েছে, তবে অবিলম্বে এটি একটি শক্তিশালী এবং অনন্য পাসওয়ার্ডে পরিবর্তন করুন যা আপনি অন্য কোথাও ব্যবহার করছেন না।

২. দ্বি-স্তরীয় প্রমাণীকরণ চালু করুন: দ্বি-স্তরীয় প্রমাণীকরণ চালু করে আপনার অ্যাকাউন্টে একটি অতিরিক্ত নিরাপত্তা স্তর যোগ করুন, যা আক্রমণকারীদের জন্য অ্যাক্সেস পাওয়া আরও কঠিন করে তোলে।

৩. আপনার অ্যাকাউন্টের কার্যকলাপ পরীক্ষা করুন: কেউ লগ ইন করার বা আপনার সেটিংস পরিবর্তন করার চেষ্টা করার মতো কোনো অস্বাভাবিক বা অননুমোদিত কার্যকলাপ হয়েছে কিনা তা দেখতে আপনার অ্যাকাউন্টের কার্যকলাপ লগ দেখুন।

৪. আপনার নিরাপত্তা প্রশ্ন আপডেট করুন: যদি আপনার অ্যাকাউন্টের সাথে নিরাপত্তা প্রশ্ন যুক্ত থাকে, তবে সেগুলো আপডেট করুন, কারণ সেগুলোও আপস করা হয়ে থাকতে পারে।

৫. আপনার অন্যান্য অ্যাকাউন্ট পরীক্ষা করুন: আপনি অন্য কোনো অ্যাকাউন্টের জন্য একই বা অনুরূপ পাসওয়ার্ড ব্যবহার করেছেন কিনা তা পরীক্ষা করুন এবং সেগুলো শক্তিশালী ও অনন্য পাসওয়ার্ডে আপডেট করুন।

৬. আপনার আর্থিক বিষয়ের প্রতি নজর রাখুন: আপনি চেনেন না বা অনুমোদন করেননি এমন কোনো লেনদেনের জন্য আপনার ব্যাঙ্ক এবং ক্রেডিট কার্ড অ্যাকাউন্ট পরীক্ষা করুন।

৭. লঙ্ঘনটি রিপোর্ট করুন: ডেটা লঙ্ঘন সম্পর্কে সংশ্লিষ্ট কর্তৃপক্ষকে জানান, তা সে আপসকৃত অ্যাকাউন্টের মালিক কোম্পানি হোক বা কোনো সরকারি সংস্থা।

৮. একটি পাসওয়ার্ড ম্যানেজার ব্যবহার করুন: আপনার সমস্ত অ্যাকাউন্টের জন্য শক্তিশালী, অনন্য পাসওয়ার্ড তৈরি ও সংরক্ষণ করতে একটি পাসওয়ার্ড ম্যানেজার ব্যবহার করার কথা বিবেচনা করুন।

৯. সতর্ক থাকুন: পরিচয় চুরি বা প্রতারণামূলক কার্যকলাপের কোনো লক্ষণের প্রতি সজাগ থাকুন, এবং সন্দেহজনক কিছু অবিলম্বে রিপোর্ট করুন।

১০. অনলাইন নিরাপত্তা সম্পর্কে আরও জানুন: অনলাইন নিরাপত্তার সর্বোত্তম অনুশীলন সম্পর্কে নিজেকে শিক্ষিত করুন এবং সর্বশেষ হুমকি ও দুর্বলতা সম্পর্কে অবগত থাকুন।

আপনার জন্য অনুসন্ধানের ফলাফল!

চমৎকার! এই পাসওয়ার্ডটি নিরাপদ।

তবুও আমরা আপনাকে পাসওয়ার্ড সুরক্ষিত রাখার ব্যাপারে সতর্কতা অবলম্বন করার অনুরোধ করছি। অনুগ্রহ করে পাসওয়ার্ড সর্বোত্তম অনুশীলন পড়ুন এবং নিরাপদ থাকুন।
ওহো! অনুগ্রহ করে একটি পাসওয়ার্ড দিয়ে আবার অনুসন্ধান করার চেষ্টা করুন।

মনে হচ্ছে পরীক্ষা করার আগে আপনি টাইপ করতে ভুলে গেছেন
